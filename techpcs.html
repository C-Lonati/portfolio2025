//배너 오른쪽->왼쪽 이동 함수
let bannerRight = function() {
if(!banner) return 0;
$bannerNow = $('#banner .img' + $bannerIndex);
$bannerIndex++;
if ($bannerIndex > $bannerLen) {
    $bannerIndex = 1;
    $bannerNext = $('#banner .img' + $bannerIndex);
    $bannerNext.css('left', '100%');
    } else {
    $bannerNext = $('#banner .img' + $bannerIndex);
    $bannerNext.css('left', '100%');
    }
    $bannerNow.animate({
    'left': '0',
    'left': '-100%',
    }, 1000);
    $bannerNext.animate({
    'left': '100%',
    'left': '0',
    }, 999);
}

$(document).on('click','#classification li', function(){
    let companyOk = $(this).parent().parent().hasClass('company')
    if($(this).hasClass('liSelect')){
        $(this).removeClass('liSelect');
        if(companyOk){
            $('.partsList>li').hide();
            for(let i = 0; i < $('.partsList>li').length;i++){
                for(let j = 0; j < getData.length; j++) {
                    if (getData[j] == $(this).text()) { //누른 버튼과 일치하는 텍스트 찾기
                        getData.splice(j, 1); //일치하는 텍스트 삭제
                        j--; //배열 인덱스넘버 재정렬
                    }
                    if ($('.partsList>li').eq(i).hasClass(getData[j])) { 
                        $('.partsList>li').eq(i).show();
                    }
                }
            }
        }
        if(getData.length == 0) $('.partsList>li').show();
    }else {
        $(this).addClass('liSelect');
        if(companyOk){
            getData[getData.length] = $(this).text();
            $('.partsList>li').hide();
            for(let i = 0; i < $('.partsList>li').length;i++){
                for(let j = 0; j < getData.length; j++) {
                    if ($('.partsList>li').eq(i).hasClass(getData[j])) { 
                        $('.partsList>li').eq(i).show();
                    }
                }
            }
        }
    }
});

구현할 때 생긴 문제점

1. 비동기 로딩 시 문제점
JQuery라이브러리의 document.load()를 활용해 비교적 쉽게 비동기 로딩에 성공했다. 비동기 로딩을 사용하려면 해당 파일이 서버에 업로드 되어 있어야 했고, 사소한 수정사항(html 구조 수정)이 발생했을 때 서버에 업로드 하고 적용되기까지 기다려야 해서 수정 및 피드백 시간이 상대적으로 오래 걸렸다.
요소에 이벤트 바인딩을 할때 기존에 사용하던 document.on('click',()=>{})방식이 적용되지 않았다. 사용하던 JQuery 버전이 3.6.0버전이라 .live()는 사용할 수 없었고, 이를 해결하기 위해 아래 방법을 사용했다.

가. document.ready.on.('click',()=>{})

해당 방법을 사용하자 비동기 로딩한 요소에 대한 이벤트 바인딩이 정상적으로 적용되는 것처럼 보였다. 하지만 이벤트 바인딩이 작동하지 않는 경우가 지속적으로 발생해 다른 방법을 찾아보았다. 나중에 알아본 바로는 구버전에서 사용하던 문법의 잔재가 남아 있는 경우라 작동에 대한 보장이 되지 않았다.

나. $(document).on('click',element,()=>{})

해당 방법을 사용하자 이벤트 바인딩이 정상적으로 작동했고 문제를 해결했다.
2. 데이터 보관에 대한 고민
mySQL과 같은 DB를 이용할 지 data.js와 같은 파일 형태로 보관할 지에 대해 고민이 되었다.
DB구축을 하기에는 컴퓨터의 연식이 오래되고 사양이 낮아 힘들다고 판단, pcBuidData.js파일에 데이터를 보관하기로 결정했다.
대신 pcBuidData.js파일에서는 데이터를 Array 내부의 Class로 선언해 최대한 DB에서 데이터를 추출했을때와 유사한 환경을 만들었다.
3. 데이터 사용에 대한 고민
초기에는 보여줄 모든 데이터를 조합한 후에 선택된 탭만 보여주는 방식을 사용하려고 했다. 이런 방식을 사용하면 대량의 데이터를 넣었을 때, 성능에 중대한 영향을 줄 것이 우려되었다. 탭메뉴 클릭 시 그에 맞춰 데이터를 조합하는 방식으로 변경하였다.
